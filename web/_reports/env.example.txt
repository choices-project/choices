# Environment Variables Example
# Copy this file to .env.local and fill in your actual values

# Supabase Configuration (Required)
NEXT_PUBLIC_SUPABASE_URL=your_supabase_project_url
NEXT_PUBLIC_SUPABASE_ANON_KEY=your_supabase_anon_key
SUPABASE_SERVICE_ROLE_KEY=your_supabase_service_role_key

# Database Configuration (Required for production)
DATABASE_URL=postgresql://username:password@host:port/database

# API Keys (Optional - for external services)
GOOGLE_CIVIC_INFO_API_KEY=your_google_civic_info_api_key
PROPUBLICA_API_KEY=your_propublica_api_key

# Security Configuration (Required for production)
NEXTAUTH_SECRET=your_nextauth_secret_key
NEXTAUTH_URL=http://localhost:3000

# Development Configuration (Optional)
NODE_ENV=development
NEXT_PUBLIC_APP_ENV=development

# Feature Flags (Optional - defaults to false)
NEXT_PUBLIC_ENABLE_ADMIN_PANEL=false
NEXT_PUBLIC_ENABLE_ANALYTICS=false
NEXT_PUBLIC_ENABLE_PWA=false

# Rate Limiting (Optional - defaults to standard values)
RATE_LIMIT_REQUESTS_PER_MINUTE=60
RATE_LIMIT_BURST_SIZE=10

# Logging (Optional)
LOG_LEVEL=info
ENABLE_DEBUG_LOGGING=false

# Performance (Optional)
ENABLE_PERFORMANCE_MONITORING=false
CACHE_TTL_SECONDS=300

# Privacy (Optional)
DEFAULT_PRIVACY_LEVEL=standard
ENABLE_DIFFERENTIAL_PRIVACY=false
